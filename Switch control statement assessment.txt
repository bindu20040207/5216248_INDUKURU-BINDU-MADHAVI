#include <stdio.h>
int main() {
    int day,n;
    for (int i = 0; i < 10; i++) {
        printf("Enter a number (1-7) to get the corresponding day of the week:\n");
        scanf("%d",&n);
        scanf("%d", &day);
        switch (day) {
            case 1:
                printf("Monday\n");
                return 0;
            case 2:
                printf("Tuesday\n");
                return 0;
            case 3:
                printf("Wednesday\n");
                return 0;
            case 4:
                printf("Thursday\n");
                return 0;
            case 5:
                printf("Friday\n");
                return 0;
            case 6:
                printf("Saturday\n");
                return 0;
            case 7:
                printf("Sunday\n");
                return 0;
            default:
                printf("Invalid input! Please enter a number between 1 and 7.\n");
        }
    }
    printf(" Invalid number please enter valid number\n");
    return 0;
}
