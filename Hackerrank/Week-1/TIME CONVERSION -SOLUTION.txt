def timeConversion(s):
    # Write your code here
    period = s[-2:]
    hour = int(s[:2])
    if period == "AM":
        if hour == 12:
            hour = 0
    else:
        if hour != 12:
            hour += 12
    return "{:02}:{:s}".format(hour, s[3:-2])


if __name__ == '__main__':
    fptr = open(os.environ['OUTPUT_PATH'], 'w')

    s = input()

    result = timeConversion(s)

    fptr.write(result + '\n')

    fptr.close()
